3       public String recomendarActividad(
4               boolean plenasFacultades,
5               boolean sintomasRecientes,
6               double temperatura,
7               double humedad,
8               String precipitacion,
9               boolean superaAforo,
10              boolean nublado
11      ) {
12  
13          if (!plenasFacultades || sintomasRecientes) {
14              return "Ninguna actividad";
15          }
16  
17          if (temperatura < 0 && humedad < 15 &&
18                  (precipitacion.equals("nieve") || precipitacion.equals("agua"))) {
19              return "Quedarse en casa";
20          }
21  
22          if (temperatura < 0 && humedad < 15 && precipitacion.equals("ninguna")) {
23              if (!superaAforo) return "Esquiar";
24              else return "Actividad no disponible por aforo";
25          }
26  
27          if (temperatura >= 0 && temperatura < 15 &&
28                  precipitacion.equals("ninguna")) {
29              if (!superaAforo) return "Senderismo o escalada";
30              else return "Actividad no disponible por aforo";
31          }
32  
33          if (temperatura >= 15 && temperatura < 25 &&
34                  precipitacion.equals("ninguna") &&
35                  !nublado &&
36                  humedad <= 60) {
37              if (!superaAforo) return "Actividades de primavera/verano/otoño";
38              else return "Actividad no disponible por aforo";
39          }
40  
41          if (temperatura >= 25 && temperatura <= 35 &&
42                  precipitacion.equals("ninguna")) {
43              return "Culturales o gastronómicas";
44          }
45  
46          if (temperatura > 30 && precipitacion.equals("ninguna")) {
47              if (!superaAforo) return "Playa o piscina";
48              else return "Actividad no disponible por aforo";
49          }
50  
51          return "Sin recomendación";
52      }
